    • Router

    JavaScript library for client-side routing

    Project maintained by dmkuznetsov Hosted on GitHub Pages — Theme by mattgraham

    Example:

    • main
    • hello, John
    • hello, John Do
    • not found

    Result: 

    var showResult = function (text) {
    document.getElementById('result').innerHTML = text;
}

var router = new Router('!');
router
    .when('/', function () {
        showResult('Main page. Request: ' + JSON.stringify(this.request));
    })
    .when('/hello/:name', function (name) {
        showResult('Hello, ' + name + '! Request: ' + JSON.stringify(this.request));
    })
    // we can use get values as arguments
    // or we can use this.request in Route context
    .when('/hello/:name.:surname', function (name, surname) {
        showResult('My name is ' + name + ' ' + surname + '! Request: ' + JSON.stringify(this.request));
    })
    // otherwise function haven't route context, so this.request will be undefined
    .otherwise(function () {
        showResult('Page not found')
    })
    .listen();
